204  NodeOps backwardOps() {
205  // For each row, the Jacobian times vector is given by:
206  // J * dy = p .* (dy - avg*1)
207  // where avg = p'*dy and p is the softmax output (probabilities).
208  //
209  // For more information, see sec. 2.5 of the following reference:
210  // André F. T. Martins and Ramon Astudillo.
211  // "From Softmax to Sparsemax: A Sparse Model of Attention and Multi-Label
212  // Classification." ICML 2016.
213  // http://jmlr.org/proceedings/papers/v48/martins16.pdf
214 
215  // val_ is already masked if there is a mask, so no need to apply here.
216 
217  return {NodeOp(SoftmaxGrad(child(0)->grad(), adj_, val_))};
218  }
