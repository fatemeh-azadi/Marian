203 Ptr<Scorer> scorerByType(std::string fname,

205  std::string model,

207  std::string type = options->get<std::string>("type");
208 
209  if(type == "s2s") {
210  return New<ScorerWrapper<S2S>>(fname, weight, model, options);
211  } else if(type == "amun") {

214  return New<ScorerWrapper<HardAtt>>(fname, weight, model, options);

217  } else if(type == "multi-s2s") {
218  return New<ScorerWrapper<MultiS2S>>(fname, weight, model, options);
219  } else if(type == "multi-hard-att") {
220  return New<ScorerWrapper<MultiHardSoftAtt>>(fname, weight, model, options);
221  } else {
222  UTIL_THROW2("Unknown decoder type: " + type);
223  }
