70 // the constructor just launches some amount of workers
71 inline ThreadPool::ThreadPool(size_t threads, size_t in_bound)
72  : stop(false), bound(in_bound) {
73  for (size_t i = 0;i<threads;++i)
74  workers.emplace_back(
75  [this] {
76  for(;;) {
77  std::function<void()> task;
78  {
79  std::unique_lock<std::mutex> lock(this->queue_mutex);
80  this->condition.wait(lock,
81  [this]{ return this->stop || !this->tasks.empty(); });
82  if (this->stop && this->tasks.empty())
83  return;
84  task = std::move(this->tasks.front());
85  this->tasks.pop();
86  }
87  this->bounded_condition.notify_one();
88 
89  task();
90  }
91  }
92  );
93 }

96 template<class F, class... Args>
97 auto ThreadPool::enqueue(F&& f, Args&&... args)
98  -> std::future<typename std::result_of<F(Args...)>::type>
99 {
100  using return_type = typename std::result_of<F(Args...)>::type;
101 
102  auto task = std::make_shared< std::packaged_task<return_type()> >(
103  std::bind(std::forward<F>(f), std::forward<Args>(args)...)
104  );
105 
106  std::future<return_type> res = task->get_future();
107  {
108  std::unique_lock<std::mutex> lock(queue_mutex);
109  this->bounded_condition.wait(lock, [this] { return this->tasks.size() < this->bound || this->bound == 0 || this->stop; });
110  // don't allow enqueueing after stopping the pool
111  if (stop) {
112  throw std::runtime_error("enqueue on stopped ThreadPool");
113  }
114 
115  tasks.emplace([task](){ (*task)(); });
116  }
117  condition.notify_one();
118  return res;
119 }
