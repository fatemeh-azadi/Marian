325

328 
329  // start locking

331 
332  if (threadIncr_ < cpuThreads) {

334  ret.threadInd = threadIncr_;
335  }
336  else if (threadIncr_ < cpuThreads + totGPUThreads) {
337  ret.deviceType = GPUDevice;
338  size_t threadIncr = threadIncr_ - cpuThreads;
339 
340  ret.threadInd = threadIncr / gpuDevices.size();
341 
342  size_t deviceInd = threadIncr % gpuDevices.size();
343  assert(deviceInd < gpuDevices.size());
344  ret.deviceId = gpuDevices[deviceInd];
345  }
346  else if (threadIncr_ < cpuThreads + totGPUThreads + totFPGAThreads) {
347  ret.deviceType = FPGADevice;
348  size_t threadIncr = threadIncr_ - cpuThreads - totGPUThreads;
349 
350  ret.threadInd = threadIncr / fpgaDevices.size();
351 
352  size_t deviceInd = threadIncr % fpgaDevices.size();
353  assert(deviceInd < fpgaDevices.size());
354  ret.deviceId = fpgaDevices[deviceInd];
355  }
356  else {
