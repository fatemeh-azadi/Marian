117 void OutputRec(const YAML::Node node, YAML::Emitter& out) {
118  std::set<std::string> flow = { "devices" };
119  std::set<std::string> sorter;
120  switch (node.Type()) {
121  case YAML::NodeType::Null:
122  out << node; break;
123  case YAML::NodeType::Scalar:
124  out << node; break;
125  case YAML::NodeType::Sequence:
126  out << YAML::BeginSeq;
127  for(auto&& n : node)
128  OutputRec(n, out);
129  out << YAML::EndSeq;
130  break;
131  case YAML::NodeType::Map:
132  for(auto& n : node)
133  sorter.insert(n.first.as<std::string>());
134  out << YAML::BeginMap;
135  for(auto& key : sorter) {
136  out << YAML::Key;
137  out << key;
138  out << YAML::Value;
139  if(flow.count(key))
140  out << YAML::Flow;
141  OutputRec(node[key], out);
142  }
143  out << YAML::EndMap;
144  break;
145  case YAML::NodeType::Undefined:
146  out << node; break;
147  }
148 }
