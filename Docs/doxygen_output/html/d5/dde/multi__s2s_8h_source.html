141  virtual Ptr<DecoderState> step(Ptr<ExpressionGraph> graph,
142  Ptr<DecoderState> state) {
143  using namespace keywords;
144 
145  auto embeddings = state->getTargetEmbeddings();
146 
147  // dropout target words
148  float dropoutTrg = inference_ ? 0 : opt<float>("dropout-trg");
149  if(dropoutTrg) {
150  int trgWords = embeddings->shape()[2];
151  auto trgWordDrop = graph->dropout(dropoutTrg, {1, 1, trgWords});
152  embeddings = dropout(embeddings, mask = trgWordDrop);
153  }
154 
155  if(!rnn_)
156  rnn_ = constructDecoderRNN(graph, state);
157 
158  // apply RNN to embeddings, initialized with encoder context mapped into
159  // decoder space
160  auto decoderContext = rnn_->transduce(embeddings, state->getStates());
161 
162  // retrieve the last state per layer. They are required during translation
163  // in order to continue decoding for the next word
164  rnn::States decoderStates = rnn_->lastCellStates();
165 
166  // retrieve all the aligned contexts computed by the attention mechanism
167  auto att1 = rnn_->at(0)->as<rnn::StackedCell>()->at(1)->as<rnn::Attention>();
168  auto alignedContext1 = att1->getContext();
169 
170  auto att2 = rnn_->at(0)->as<rnn::StackedCell>()->at(2)->as<rnn::Attention>();
171  auto alignedContext2 = att2->getContext();
172 
173  auto alignedContext = concatenate({alignedContext1,
174  alignedContext2},
175  axis=1);
176 
177  // construct deep output multi-layer network layer-wise
178  auto layer1 = mlp::dense(graph)
179  ("prefix", prefix_ + "_ff_logit_l1")
180  ("dim", opt<int>("dim-emb"))
181  ("activation", mlp::act::tanh)
182  ("layer-normalization", opt<bool>("layer-normalization"));
183 
184  int dimTrgVoc = opt<std::vector<int>>("dim-vocabs").back();
185 
186  auto layer2 = mlp::dense(graph)
187  ("prefix", prefix_ + "_ff_logit_l2")
188  ("dim", dimTrgVoc);
189  if(opt<bool>("tied-embeddings"))
190  layer2.tie_transposed("W", prefix_ + "_Wemb");
191 
192  // assemble layers into MLP and apply to embeddings, decoder context and
193  // aligned source context
194  auto logits = mlp::mlp(graph)
195  .push_back(layer1)
196  .push_back(layer2)
197  ->apply(embeddings, decoderContext, alignedContext);
198 
199  // return unormalized(!) probabilities
200  return New<DecoderState>(decoderStates, logits, state->getEncoderState());
201  }
