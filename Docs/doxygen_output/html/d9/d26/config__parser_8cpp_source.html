31 uint16_t guess_terminal_width(uint16_t max_width) {
32  uint16_t cols = 0;
33 #ifdef TIOCGSIZE
34  struct ttysize ts;
35  ioctl(STDIN_FILENO, TIOCGSIZE, &ts);
36  if(ts.ts_cols != 0)
37  cols = ts.ts_cols;
38 #elif defined(TIOCGWINSZ)
39  struct winsize ts;
40  ioctl(STDIN_FILENO, TIOCGWINSZ, &ts);
41  if(ts.ws_col != 0)
42  cols = ts.ws_col;
43 #endif
44  if(cols == 0) // couldn't determine terminal width
45  cols = po::options_description::m_default_line_length;
46  return max_width ? std::min(cols, max_width) : cols;
47 }
